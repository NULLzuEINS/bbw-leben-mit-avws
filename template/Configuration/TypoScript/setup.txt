# Language menu
# https://docs.typo3.org/typo3cms/FrontendLocalizationGuide/LanguageMenu/Index.html
lib.languageMenu = HMENU
lib.languageMenu {
    special = language
    special.value = 0,1
    special.normalWhenNoLanguage = 0
    1 = TMENU
    1 {
        noBlur = 1
        NO = 1
        NO {
            doNotLinkIt = 1
            linkWrap = <li>|</li>

            stdWrap.override = Text || Deutsche Gebärdensprache || Lautunterstützte Gebärdensprache
            stdWrap {
                typolink {
                    parameter.data = page:uid
                    additionalParams = &L=0 || &L=1 || &L=2
                    addQueryString = 1
                    addQueryString.exclude = L,id,no_cache
                    addQueryString.method = GET
                    no_cache = 0
                }
            }
        }
        ACT < .NO
        USERDEF1 < .NO
        USERDEF2 < .ACT
    }
}

lib.languageMenuDesktop = COA
lib.languageMenuDesktop {
    10  < lib.languageMenu
    10 {
        wrap = <div class="container"><ul class="nav nav-pills hidden-xs hidden-sm pull-right" style="display: inline-block;">|</ul></div>
        1 {
            ACT.linkWrap = <li class="active">|</li>
            USERDEF1 {
                linkWrap = <li class="disabled">|</li>

            }
            USERDEF2 {
                linkWrap = <li class="disabled">|</li>
                stdWrap.typolink.ATagParams=class="disabled"
            }
        }
    }
}
